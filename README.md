# Proga2moon
UNIT.HACK 2025. Кейс приложение/бот от NAUMEN


# Структура проекта

---

```
Proga2moon/
    │
    ├── bot.py                  # Главный файл для запуска бота
    ├── config.py               # Конфигурационные параметры (токен, настройки БД и т.д.)
    ├── database/               # Работа с базой данных
    │   ├── __init__.py
    │   ├── models.py           # Модели данных (пользователи, мероприятия, записи)
    │   ├── crud.py             # CRUD-операции (создание, чтение, обновление, удаление)
    │   └── database.py         # Инициализация и подключение к БД
    │
    ├── handlers/               # Обработчики сообщений и команд
    │   ├── __init__.py
    │   ├── start.py            # Команда /start и приветствие
    │   ├── events.py           # Управление мероприятиями и мастер-классами
    │   ├── registration.py     # Запись на мероприятия
    │   ├── admin.py            # Админ-функции (рассылки, управление событиями)
    │   └── common.py           # Общие команды (help, info, feedback)
    │
    ├── keyboards/              # Клавиатуры и inline-кнопки
    │   ├── __init__.py
    │   ├── main_menu.py        # Главное меню
    │   ├── event_keyboards.py  # Кнопки для мероприятий
    │   └── admin_keyboards.py  # Кнопки для админов
    │
    ├── services/               # Дополнительные сервисы
    │   ├── __init__.py
    │   ├── notifications.py    # Уведомления и рассылки
    │   ├── utils.py            # Вспомогательные функции
    │   └── waiting_list.py     # Управление листом ожидания
    │
    ├── locales/                # Локализация (если нужно)
    │   ├── en.json
    │   └── ru.json
    │
    └── pyproject.toml          # Зависимости
```
